    const LIFF_ID = '2010107887-oOC80iWL';


let lineProfile = {

  userId: '',

  displayName: ''

};


async function initLiff() {

  const box = document.getElementById('lineInfo');


  try {

    if (box) box.innerHTML = 'LIFF初期化中...';


    await liff.init({ liffId: LIFF_ID });


    if (box) {

      box.innerHTML =

        'LIFF初期化OK<br>' +

        'LINE内: ' + liff.isInClient() + '<br>' +

        'ログイン: ' + liff.isLoggedIn();

    }


    const profile = await liff.getProfile();


    lineProfile.userId = profile.userId;

    lineProfile.displayName = profile.displayName;


    if (box) {

      box.innerHTML =

        'LINE名：' + lineProfile.displayName + '<br>' +

        'LINE ID取得：OK';

    }


  } catch (err) {

    if (box) {

      box.innerHTML = 'LIFFエラー：' + err.message;

    }

    console.error(err);

  }

}

    let products = [];

    let cart = {};


    const emojiMap = {

      'にんじん': '🥕',

      'キャベツ': '🥬',

      '玉ねぎ': '🧅',

      'じゃがいも': '🥔',

      'ほうれん草': '🥬',

      '大根': '🥬',

      'トマト': '🍅',

      'きゅうり': '🥒',

      'なす': '🍆',

      'とうもろこし': '🌽',

      'オクラ': '🌿'

    };


    function loadProducts() {

      google.script.run

        .withSuccessHandler(data => {

          products = data;

          renderProducts();

        })

        .withFailureHandler(err => {

          document.getElementById('productList').innerHTML =

            '<p style="color:red;">商品を読み込めませんでした。</p>';

          console.error(err);

        })

        .getProducts();

    }


    function getEmoji(name) {

      for (const key in emojiMap) {

        if (name.includes(key)) return emojiMap[key];

      }

      return '🥬';

    }


    function renderProducts() {

      const list = document.getElementById('productList');


      if (!products.length) {

        list.innerHTML = '<p>現在、予約できる商品がありません。</p>';

        return;

      }


      list.className = '';


      list.innerHTML = products.map(p => {

        cart[p.id] = cart[p.id] || 0;


        return `

          <div class="product">

            <div class="thumb">${getEmoji(p.name)}</div>


            <div class="info">

              <div class="name">${p.name}</div>

              <div class="price">${p.price.toLocaleString()}円 / ${p.unit}</div>

              <div class="stock">在庫：${p.stock}</div>

            </div>


            <div class="counter">

              <button class="count-btn" onclick="changeQty('${p.id}', -1)">−</button>

              <div class="count" id="count-${p.id}">${cart[p.id]}</div>

              <button class="count-btn" onclick="changeQty('${p.id}', 1)">＋</button>

            </div>

          </div>

        `;

      }).join('');

    }


    function changeQty(id, amount) {

      const product = products.find(p => p.id === id);

      const current = cart[id] || 0;

      const next = current + amount;


      if (next < 0) return;

      if (next > product.stock) {

        alert('在庫数を超えています');

        return;

      }


      cart[id] = next;

      document.getElementById(`count-${id}`).textContent = next;

      updateTotal();

    }


    function updateTotal() {

      let total = 0;


      products.forEach(p => {

        total += p.price * (cart[p.id] || 0);

      });


      document.getElementById('totalPrice').textContent =

        total.toLocaleString() + '円';

    }


    function goNext() {

  const selected = products

    .filter(p => cart[p.id] > 0)

    .map(p => ({

      id: p.id,

      name: p.name,

      price: p.price,

      unit: p.unit,

      qty: cart[p.id],

      subtotal: p.price * cart[p.id]

    }));


  if (!selected.length) {

    alert('商品を1つ以上選んでください');

    return;

  }


  const total = selected.reduce((sum, item) => sum + item.subtotal, 0);


  const formHtml = `

    <div class="form-area" id="formArea">

      <h2>お客様情報</h2>


      <div class="selected-box">

        <h3>ご予約内容</h3>

        ${selected.map(item => `

          <div class="selected-row">

            <span>${item.name} × ${item.qty}</span>

            <strong>${item.subtotal.toLocaleString()}円</strong>

          </div>

        `).join('')}

        <div class="selected-total">

          合計 ${total.toLocaleString()}円

        </div>

      </div>


      <label>お名前 <span>必須</span></label>

      <input type="text" id="customerName" placeholder="例）山田 太郎">


      <label>電話番号 <span>必須</span></label>

      <input type="tel" id="customerPhone" placeholder="例）090-1234-5678">


      <label>LINE名</label>

      <input type="text" id="lineName" placeholder="例）たろう">


      <label>備考</label>

      <textarea id="note" placeholder="ご要望があれば入力してください"></textarea>


      <button class="submit-btn" onclick="submitReservation()">予約を確定する</button>

    </div>

  `;


  document.getElementById('productList').insertAdjacentHTML('afterend', formHtml);

  document.getElementById('formArea').scrollIntoView({ behavior: 'smooth' });

}

function submitReservation() {

  const name = document.getElementById('customerName').value.trim();

  const phone = document.getElementById('customerPhone').value.trim();

  const lineName = document.getElementById('lineName').value.trim();

  const note = document.getElementById('note').value.trim();


  if (!name) {

    alert('お名前を入力してください');

    return;

  }


  if (!phone) {

    alert('電話番号を入力してください');

    return;

  }


  const items = products

    .filter(p => cart[p.id] > 0)

    .map(p => ({

      id: p.id,

      name: p.name,

      price: p.price,

      unit: p.unit,

      qty: cart[p.id],

      subtotal: p.price * cart[p.id]

    }));


  const total = items.reduce((sum, item) => sum + item.subtotal, 0);


  const order = {

  name,

  phone,

  lineUserId: lineProfile.userId,

  lineDisplayName: lineProfile.displayName,

  lineName,

  note,

  items,

  total

};


  google.script.run

    .withSuccessHandler(res => {

      alert('予約が完了しました。\n予約番号：' + res.reservationId);

      location.reload();

    })

    .withFailureHandler(err => {

      alert('予約保存に失敗しました');

      console.error(err);

    })

    .saveReservation(order);

}
